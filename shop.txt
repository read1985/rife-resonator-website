      WebPlatform.onReady(function () {
        var root = $('#store-catalog-widget-1718370429432');
        var hasQuickView = true;
        var showBuyButtonInLists = false;
        var sort = '';
        var searchFilter = '';
        var filterUrlPart = '';
        var variationsFilter = null;
        var sortByFilter = root.find('#sortByFilter');
        var productsPerPageFilter = root.find('#productsPerPageFilter');
        var category = '';
        var lowerPriceRange = 0;
        var upperPriceRange = Math.abs(1600);
        var maxPrice = Math.abs(1600);
        var subcategories = '';
        var mobileStoreOverlay = root.find('#mobileStoreOverlay');
        var storeProductsFilterWrapper = root.find('#storeProductsFilterWrapper');
        var productFilterMobileButton = root.find('#productFilterMobileButton');
        var variations = storeProductsFilterWrapper.find('.filter-by-variation');
        var variationsFilterUrl = '';

        if (variations.length > 0) {
          variations.on('click', '.variation-filter-link', function (event) {
            var li = $(event.target).closest('li').toggleClass('active');
            var url = '';
            variations.each(function (idx, variation) {
              variation = $(variation);
              var checked = variation.find('li.active');
              if (checked.length > 0) {
                var filterValues = '';
                checked.each(function (idx, elem) {
                  var filter = ($(elem).data('filter') || '') + '';
                  if (filter) {
                    if (filterValues !== '')
                      filterValues += ',';
                    filterValues += encodeURIComponent(filter.replace('&', '^'));
                  }
                });
                if (filterValues !== '') {
                  if (url !== '')
                    url += '&';
                  url += encodeURIComponent(variation.find('h3').text().replace('&', '^'));
                  url += '=' + filterValues;
                }
              }
            });
            variationsFilterUrl = url;
            refreshFiltering();
          });
        }

        if (variationsFilter !== null) {
          variations.each(function (idx, elem) {
            var root = $(elem);
            var heading = root.find('h3').text().toLowerCase();
            var filtered = [];
            for (var idx = 0; idx < variationsFilter.length; ++idx) {
              if (variationsFilter[idx].name.toLowerCase() === heading) {
                filtered = variationsFilter[idx].values;
                break;
              }
            }
            if (filtered.length > 0) {
              var items = root.find('li');
              items.each(function (idx, li) {
                var li = $(li);
                var filter = ((li.data('filter') || '') + '').toLowerCase();
                for (var idx = 0; idx < filtered.length; ++idx) {
                  if (filtered[idx].first.toLowerCase() === filter) {
                    li.addClass('active');
                    break;
                  }
                }
              });
            }
          });
        }

        var refreshFiltering = function () {
          var newSort = sortByFilter.val();
          var newRows = +productsPerPageFilter.val();
          var newUrl = WebPlatform.Store.baseUrl;
          if (category !== '')
            newUrl += '/category/' + category;
          if (subcategories !== '')
            newUrl += '/sub/' + subcategories;
          if (variationsFilterUrl !== '')
            newUrl += '/var/' + variationsFilterUrl;
          if (newRows !== 4)
            newUrl += '/rows/' + newRows;
          if (newSort)
            newUrl += '/sort/' + newSort;
          if (searchFilter)
            newUrl += '/search/' + encodeURI(searchFilter);
          if (lowerPriceRange > 0 || upperPriceRange !== maxPrice)
            newUrl += '/range/' + lowerPriceRange + ',' + upperPriceRange;
          ClientTools.showLoader();
          window.location.href = newUrl;
        };
        sortByFilter.val(sort).add(productsPerPageFilter).change(refreshFiltering);

        try {
          root.find('#productsFilter').on('click', '.has-subcategories > a', function (event) {
            var elem = $(this);
            var clickedOnArrow = $(event.target).is('.arrow');
            var li = elem.parent();
            if (!clickedOnArrow && li.hasClass('open') && li.children('ul').children('.active').length > 0)
              return;
            if (li.hasClass('open')) {
              li.removeClass('open');
              return WebPlatform.preventDefault(event);
            }
            else
            if (li.children('ul').length > 0) {
              li.addClass('open');
              return WebPlatform.preventDefault(event);
            }
            else
            if (clickedOnArrow) {
              li.addClass('open');
              WebPlatform.makeClientWebsiteRequest({
                action: 'getStoreSubcategories',
                data: {
                  parent: elem.data('id')
                },
                onOk: function (result) {
                  var categoriesPath = [elem.data('url')];
                  var rootElem = elem.closest('ul');
                  while (true) {
                    if (rootElem.hasClass('filter-categories'))
                      break;
                    categoriesPath.push(rootElem.parent().children('a').data('url'));
                    rootElem = rootElem.parent().closest('ul');
                  }

                  var baseUrl = WebPlatform.Store.baseUrl + '/category/';
                  for (var idx = categoriesPath.length - 1; idx >= 0; --idx) {
                    if (idx === categoriesPath.length - 1)
                      baseUrl += categoriesPath[idx] + '/sub/';
                    else
                      baseUrl += categoriesPath[idx] + ',';
                  }

                  var categories = result.data;
                  var markup = '<ul>';
                  for (var idx = 0; idx < categories.length; ++idx) {
                    var category = categories[idx];
                    var hasSubcategories = category.left + 1 < category.right;
                    markup += '<li class="' + (hasSubcategories ? 'has-subcategories' : '') + '">';
                    markup += '<a data-url="' + category.url + '" data-id="' + category.id + '" href="' + baseUrl + category.url + filterUrlPart + '">' + category.name;
                    if (hasSubcategories) {
                      markup += '<div class="arrow"></div>';
                    }
                    markup += '</a>';
                    markup += '</li>';
                  }
                  markup += '</ul>';
                  elem.after(markup);
                }
              });
              return WebPlatform.preventDefault(event);
            }
          });
          var onPriceSliderChanged = $.debounce(800, function () {
            var newUrl = WebPlatform.clearUrlParameter(window.location.href, 'range', WebPlatform.URL_PARAM_ANY);
            newUrl = WebPlatform.clearUrlParameter(newUrl, 'page', WebPlatform.URL_PARAM_ANY);
            if (!newUrl.endsWith('/'))
              newUrl += '/';
            newUrl += 'range/' + lowerPriceRange + ',' + upperPriceRange;
            ClientTools.showLoader();
            window.location.href = newUrl;
          });
          var searchFilterField = $('#searchFilter-1718370429432');
          if (searchFilterField.length) {
            var doSearch = function () {
              var searchVal = searchFilterField.val().trim();
              if (searchVal !== searchFilter) {
                searchFilter = searchVal;
                refreshFiltering();
              }
            };
            searchFilterField.val(searchFilter).keypress(function (e) {
              if (e.which === 13) {
                doSearch();
              }
            });
            $('.search-widget-icon').click(doSearch);
          }
          root.find('#lowerPriceRange').val(WebPlatform.currToStr(lowerPriceRange, WebPlatform.Store.currencySymbol, WebPlatform.Store.currencySymbolAtLeft, WebPlatform.Store.currency));
          root.find('#upperPriceRange').val(WebPlatform.currToStr(upperPriceRange, WebPlatform.Store.currencySymbol, WebPlatform.Store.currencySymbolAtLeft, WebPlatform.Store.currency));
          root.find("#priceSlider").slider({
            values: [lowerPriceRange, upperPriceRange],
            min: 0,
            max: maxPrice,
            step: 5,
            range: true,
            slide: function (event, ui) {
              lowerPriceRange = ui.values[0];
              upperPriceRange = ui.values[1];
              root.find('#lowerPriceRange').val(WebPlatform.currToStr(ui.values[0], WebPlatform.Store.currencySymbol, WebPlatform.Store.currencySymbolAtLeft, WebPlatform.Store.currency));
              root.find('#upperPriceRange').val(WebPlatform.currToStr(ui.values[1], WebPlatform.Store.currencySymbol, WebPlatform.Store.currencySymbolAtLeft, WebPlatform.Store.currency));
              onPriceSliderChanged();
            }
          });
          productFilterMobileButton.click(function () {
            if (mobileStoreOverlay.hasClass('none')) {
              mobileStoreOverlay.removeClass('none');
              storeProductsFilterWrapper.show();
              $('body').addClass('with-product-filter');
            }
            else {
              mobileStoreOverlay.addClass('none');
              storeProductsFilterWrapper.hide();
              $('body').removeClass('with-product-filter');
            }
          });
          root.find('.mobile-close-btn').click(function () {
            productFilterMobileButton.click();
          });
        } catch (e) {
          Log.error(e);
        }

        if (hasQuickView || showBuyButtonInLists) {
          root.find('.products-list').on('click', '[data-role="viewProduct"]', function () {
            var id = $(this).closest('.product-item').data('id');
            WebPlatform.Store.showProductInQuickView(id);
            return false;
          });
        }

        if (hasQuickView || showBuyButtonInLists) {
          root.find('.products-list').on('click', '[data-role="addToCart"]', function () {
            var elem = $(this);
            var product = elem.closest('.product-item');
            var id = product.data('id');
            var process = elem.data('process');
            WebPlatform.Store.addProductToCart(id, process);
            return false;
          });
        }
      });
    </script>
